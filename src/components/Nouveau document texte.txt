import React, { useState, useEffect } from 'react';
import LanguageSwitcher from './LanguageSwitcher';

function AdminPanel({ user, onLogout }) {
    const [requests, setRequests] = useState([]);
    const [loading, setLoading] = useState(true);
    const [selectedDuration, setSelectedDuration] = useState({});

    useEffect(() => {
        fetchRequests();
    }, []);

    const fetchRequests = async () => {
        try {
            const res = await fetch('https://chatalgerie-backend.onrender.com/api/admin/payment-requests');
            const data = await res.json();
            setRequests(data);
            setLoading(false);
        } catch (error) {
            console.error('❌ خطأ:', error);
            setLoading(false);
        }
    };

    const handleApprove = async (requestId, userId) => {
        const duration = selectedDuration[requestId] || 1;
        if (!window.confirm(`هل تريد تفعيل الاشتراك لمدة ${duration} شهر؟`)) return;

        try {
            const res = await fetch('https://chatalgerie-backend.onrender.com/api/admin/approve-payment', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ requestId, userId, duration })
            });
            const result = await res.json();
            alert(result.message);
            fetchRequests();
        } catch (error) {
            alert('❌ خطأ في الخادم');
        }
    };

    const handleReject = async (requestId) => {
        if (!window.confirm('هل تريد رفض هذا الطلب؟')) return;

        try {
            const res = await fetch('https://chatalgerie-backend.onrender.com/api/admin/reject-payment', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ requestId })
            });
            const result = await res.json();
            alert(result.message);
            fetchRequests();
        } catch (error) {
            alert('❌ خطأ في الخادم');
        }
    };

    if (!user?.isAdmin) {
        return (
            <div style={styles.container}>
                <div style={styles.card}>
                    <h1>⛔ غير مصرح لك بالدخول</h1>
                    <p>هذه الصفحة مخصصة للمسؤولين فقط.</p>
                    <button onClick={onLogout} style={styles.logoutBtn}>🚪 تسجيل الخروج</button>
                </div>
            </div>
        );
    }

    return (
        <div style={styles.container}>
            <div style={styles.header}>
                <h1 style={styles.title}>🔐 لوحة التحكم</h1>
                <div style={styles.headerActions}>
                    <button onClick={fetchRequests} style={styles.refreshBtn}>🔄 تحديث</button>
                    <button onClick={onLogout} style={styles.logoutBtn}>🚪 تسجيل الخروج</button>
                </div>
            </div>

            {loading ? (
                <p style={styles.loading}>⏳ جاري التحميل...</p>
            ) : (
                <div style={styles.requestsContainer}>
                    {requests.length === 0 ? (
                        <p style={styles.empty}>📭 لا توجد طلبات دفع</p>
                    ) : (
                        requests.map((req) => (
                            <div key={req.id} style={styles.requestCard}>
                                <div style={styles.requestInfo}>
                                    <p><strong>👤 {req.username}</strong></p>
                                    <p>📧 {req.useremail}</p>
                                    <p>📱 {req.phone}</p>
                                    <p>💰 {req.price} DA</p>
                                    <p>📅 {req.plan}</p>
                                    <p>📝 {req.reference || 'بدون مرجع'}</p>
                                    <p>📌 الحالة: <span style={{ 
                                        color: req.status === 'pending' ? 'orange' : 
                                               req.status === 'approved' ? 'green' : 'red',
                                        fontWeight: 'bold'
                                    }}>{req.status === 'pending' ? '⏳ قيد المراجعة' : 
                                        req.status === 'approved' ? '✅ تم الموافقة' : '❌ مرفوض'}</span></p>
                                    {req.receiptimage && (
                                        <a href={req.receiptimage} target="_blank" rel="noopener noreferrer">📷 عرض الإيصال</a>
                                    )}
                                </div>
                                {req.status === 'pending' && (
                                    <div style={styles.actions}>
                                        <select
                                            value={selectedDuration[req.id] || 1}
                                            onChange={(e) => setSelectedDuration({ ...selectedDuration, [req.id]: parseInt(e.target.value) })}
                                            style={styles.select}
                                        >
                                            <option value={1}>1 شهر</option>
                                            <option value={3}>3 أشهر</option>
                                            <option value={6}>6 أشهر</option>
                                            <option value={12}>سنة</option>
                                            <option value={-1}>غير محدود</option>
                                        </select>
                                        <button onClick={() => handleApprove(req.id, req.userid)} style={styles.approveBtn}>✅ قبول</button>
                                        <button onClick={() => handleReject(req.id)} style={styles.rejectBtn}>❌ رفض</button>
                                    </div>
                                )}
                            </div>
                        ))
                    )}
                </div>
            )}
        </div>
    );
}
